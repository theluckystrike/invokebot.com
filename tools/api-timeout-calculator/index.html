Fetch (JavaScript)
const fetchWithTimeout = (url, options = {}, timeout = 1100) => { return Promise.race([ fetch(url, options), new Promise((_, reject) => setTimeout(() => reject(new Error('Timeout')), timeout) ) ]); }; async function callWithRetry(url, options, maxRetries = 3) { for (let i = 0; i <= maxRetries; i++) { try { return await fetchWithTimeout(url, options); } catch (e) { if (i === maxRetries) throw e; const delay = Math.pow(2, i) * 1000; await new Promise(r => setTimeout(r, delay)); } } }
